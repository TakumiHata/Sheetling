```json
  "corrections": [

    {"action": "fix_text",    "page": 85, "row": <現在行>, "col": <現在列>, "new_row": <正行>, "new_col": <正列>},
    {"action": "add_border",  "page": 85, "row": <開始行>, "end_row": <終了行>, "col": <開始列>, "end_col": <終了列>, "borders": {"top": true, "bottom": true, "left": true, "right": true}},
    {"action": "remove_border","page": 85, "row": <開始行>, "end_row": <終了行>, "col": <開始列>, "end_col": <終了列>}
  ]
}
```
